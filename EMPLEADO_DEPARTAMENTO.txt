--TABLA DEPARTAMENTO
DROP TABLE DEPARTAMENTO;
CREATE TABLE DEPARTAMENTO 
    ( 
     ID_dep     NUMBER (3)  NOT NULL , 
     nombre_dep VARCHAR2 (100)  NOT NULL 
    ) 
;

ALTER TABLE DEPARTAMENTO 
    ADD CONSTRAINT DEPARTAMENTO_PK PRIMARY KEY ( ID_dep ) ;


--TABLA EMPLEADO
DROP TABLE EMPLEADO2;
CREATE TABLE EMPLEADO2 
    ( 
     ID_emp              NUMBER  NOT NULL , 
     nombre_emp          VARCHAR2 (100)  NOT NULL , 
     ID_dep NUMBER (3)   NOT NULL 
    ) 
;

ALTER TABLE EMPLEADO2 
    ADD CONSTRAINT EMPLEADO2_PK PRIMARY KEY ( ID_emp ) ;

ALTER TABLE EMPLEADO2 
    ADD CONSTRAINT EMPLEADO2_DEPARTAMENTO_FK FOREIGN KEY 
    ( 
     ID_dep
    ) 
    REFERENCES DEPARTAMENTO 
    ( 
     ID_dep
    ) 
;
